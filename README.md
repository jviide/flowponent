# flowponent

```js
import { render } from "preact";
import { flowponent } from "flowponent";

const App = flowponent(function*(step) {
  let count = 0;

  for (;;) {
    count += yield (
      <div>
        <div>current value: {count}</div>
        <button onClick={step(1)}>+1</button>
        <button onClick={step(-1)}>-1</button>
      </div>
    );
  }
});

render(<App />, document.getElementById("root"));
```
